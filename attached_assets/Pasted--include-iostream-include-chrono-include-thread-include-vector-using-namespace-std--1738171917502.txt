using namespace std;

void delay(int milliseconds) {
    this_thread::sleep_for(chrono::milliseconds(milliseconds));
}

void typeEffect(const string& text, int speed = 50) {
    for (char c : text) {
        cout << c << flush;
        delay(speed);
    }
    cout << endl;
}
